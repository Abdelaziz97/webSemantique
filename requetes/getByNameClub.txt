SELECT ?name
WHERE{
	SELECT ?human ?name
	WHERE {?event rdf:type dbo:OlympicEvent{?event dbo:bronzeMedalist ?human. ?human dbo:club ?club . ?club foaf:name ?name } 
	UNION {?event dbo:silverMedalist ?human . ?human dbo:club ?club . ?club foaf:name ?name} 
	UNION {?event dbo:goldMedalist ?human . ?human dbo:club ?club . ?club foaf:name ?name}
	FILTER regex(?human, %ATHLETE%)
	FILTER (lang(?name) = 'en')}
	GROUP BY ?human}
