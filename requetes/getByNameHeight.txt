SELECT ?height
WHERE{
	SELECT ?height ?human
	WHERE {?event rdf:type dbo:OlympicEvent{?event dbo:bronzeMedalist ?human. ?human dbo:height ?height } 
	UNION {?event dbo:silverMedalist ?human . ?human dbo:height ?height } 
	UNION {?event dbo:goldMedalist ?human . ?human dbo:height ?height }
	FILTER regex(?human, %ATHLETE%)}
	GROUP BY ?human}
