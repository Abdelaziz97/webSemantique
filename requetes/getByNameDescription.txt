SELECT ?description
WHERE{
	SELECT ?description
	WHERE {?event rdf:type dbo:OlympicEvent{?event dbo:bronzeMedalist ?human. ?human dct:description ?description } 
	UNION {?event dbo:silverMedalist ?human . ?human dct:description ?description } 
	UNION {?event dbo:goldMedalist ?human . ?human dct:description ?description }
	FILTER regex(?human, %ATHLETE%)
        FILTER (lang(?description) = 'en')}
	GROUP BY ?human}
